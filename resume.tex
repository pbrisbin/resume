% resheader(name, email, address, phone)
\newcommand{\resheader}[4]{
    \begin{center}
        \selectfont \Huge \scshape #1 \\
        \vspace{2pt}
        \selectfont \small #2
    \end{center}

    \vspace{-7pt}
    \rule{\textwidth}{1pt}

    \vspace{-2pt}
    {\small \itshape #3 \hfill #4}
    \vspace{5pt}
}

% ressection(title)
\newcommand{\ressection}[1]{
    {\fontfamily{phv} \selectfont \large \scshape #1}
    \vspace{2pt}
}
